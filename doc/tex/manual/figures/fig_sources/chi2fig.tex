\(
 \chi^2

\)

\(\displaystyle {\hat m}_i = m_i + \sum_\alpha \Gamma_{i\alpha} \, b_\alpha \)
}

\def\CM{\mathcal{C}}

\Put(-10,25){\color{blue} \bf Full covariance matrix approach}

\Put(0,14){
\(
\chi^2 =
  \left ( \boldsymbol{\mu} - \boldsymbol{m} \right)^{\rm T}
  % + \Gamma\,\boldsymbol{b}
  \CM^{-1}
  \left( \boldsymbol{\mu} - \boldsymbol{m} \right)
  % + \boldsymbol{b}^2
\)
}

\Put(0,5){
\(
  \ \ \ \ \  =
  \displaystyle
  \sum_{ij} \left ( \mu_i - m_i \right )   
  \CM^{-1}_{ij} 
  \big ( \mu_j - m_j \big )
\)
% -------------------
\hspace*{14mm}

\(
\mathcal C = \CM^\mathrm{stat} + \CM^\mathrm{unc} + \CM^\mathrm{syst}
\)
}
